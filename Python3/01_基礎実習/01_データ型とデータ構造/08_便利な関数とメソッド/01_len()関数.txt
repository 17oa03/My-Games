#【実行結果】

>>> len([1,2,3,4,5])
5
>>> len(("small", "medium", "large"))
3
>>> data = [[1, 2], [3, 4, 5], [6, 7, 8, 9]]
>>> len(data)
3
>>> len(data[0])
2
>>> len(data[1])
3
>>> len(data[2])
4
