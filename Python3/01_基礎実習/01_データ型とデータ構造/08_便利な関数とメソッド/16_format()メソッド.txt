#【実行結果】

>>> "1 = {} 2 = {}" .format("Hello", "work")
'1 = Hello 2 = work'
>>> "value = ({}, {})" .format(2, 5)
'value = (2, 5)'
>>> "score = {}" .format(2.345)
'score = 2.345'
>>> "value = ({1}, {0})" .format(2, 5)
'value = (5, 2)'
>>> "value = ({a}, {b})" .format(a = 10.1, b = 20.2)
'value = (10.1, 20.2)'
